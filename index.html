Best-effort: se falhar, o produto fica sem link e é omitido da página (não adianta botão que não leva a nada). """ try: from integrations.shopee_affiliate import minerar_oportunidades, gerar_link_afiliado except Exception: log.info(" Shopee indisponível — usando só os links que já existem") return [p for p in produtos if p.get("link")] com_link = [] for p in produtos: if p.get("link"): com_link.append(p) continue # minera o campeão e gera o link com sub-id de rastreio (origem=bio) try: m = minerar_oportunidades(p["nome"]) if m.get("ok"): origem = m["campeao"].get("product_link") or m["campeao"].get("offer_link") if origem: sub = ["bio", re.sub(r"\W+", "", p["nome"])[:12]] lk = gerar_link_afiliado(origem, sub_ids=sub) if lk.get("ok"): p["link"] = lk["short_link"] p["titulo"] = m["campeao"].get("nome", p["titulo"]) # captura a imagem oficial do produto (pra grade visual) p["imagem"] = m["campeao"].get("imagem", "") or p.get("imagem", "") com_link.append(p) log.info(f" 🔗 link gerado: {p['nome']}") time.sleep(1.0) # respeita rate limit continue log.info(f" ⏭️ sem link pra '{p['nome']}' (omitido da página)") except Exception as e: log.warning(f" erro gerando link de '{p['nome']}': {str(e)[:50]}") return com_link # Categorias pros filtros rápidos (inferidas pelo nome do produto). # ORDEM IMPORTA: categorias mais específicas primeiro (Carro/Fitness antes de # Casa), pra "aspirador automotivo" cair em Carro, não em Casa. _CATEGORIAS_FILTRO = [ ("Carro", ("automotivo", "veicular", "carro")), ("Fitness", ("massageador", "massagem", "yoga", "pilates", "faixas", "corda", "pular", "rolo", "pistola", "cervical", "fisioterapia", "academia", "treino", "elastica", "elástica")), ("Beleza", ("modelador", "cachos", "escova", "secadora", "alisadora", "maquiagem", "skincare", "cravos", "espelho", "cabelo")), ("Cozinha", ("cortador", "legumes", "liquidificador", "balanca", "balança", "garrafa", "caneca", "termica", "térmica", "descascador", "processador", "fatiador", "espremedor", "água", "agua")), ("Tech", ("mouse", "power bank", "powerbank", "carregador", "projetor", "notebook", "cabo", "fone", "usb", "celular", "induç", "induc")), ("Casa", ("umidificador", "luminaria", "luminária", "led", "organizador", "aspirador", "tapete", "vela", "porta", "suporte")), ] def _inferir_categoria(produto: dict) -> str: """Infere a categoria do produto pelo nome/título (pros filtros).""" texto = f"{produto.get('nome','')} {produto.get('titulo','')}".lower() # ordem importa: Carro antes de Casa/Tech (mais específico) for rotulo, kws in _CATEGORIAS_FILTRO: for kw in kws: if kw in texto: return rotulo return "Outros" def _card_produto(p: dict, destaque: bool = False) -> str: """Gera o HTML de um card de produto VISUAL (foto + título + CTA), estilo LTK.""" titulo = html.escape(_titulo_legivel(p.get("titulo") or p.get("nome", ""))) nome_busca = html.escape(p.get("nome", "")) link = html.escape(p.get("link", "#")) categoria = html.escape(_inferir_categoria(p)) imagem = html.escape(p.get("imagem", "")) # Imagem do produto. Se falhar ao carregar (onerror), mostra o placeholder 🛍️ if imagem: img_html = (f'') else: img_html = "" if destaque: return f""" ⭐ Produto do vídeo de hoje
{img_html}🛍️
{titulo} 🛒 PEGAR MEU DESCONTO """ return f"""
{img_html}🛍️
{titulo} Ver na Shopee → """ def _titulo_legivel(titulo: str, limite: int = 70) -> str: """Encurta títulos gigantes de marketplace pra caber bonito no card.""" t = re.sub(r"\s+", " ", (titulo or "").strip()) if len(t) <= limite: return t return t[:limite].rsplit(" ", 1)[0] + "…" def _redes_html() -> str: """Gera os ícones de redes sociais (só os que têm URL).""" itens = [] for nome, url in REDES.items(): if url: itens.append(f'{html.escape(nome)}') if not itens: return "" return '
' + "".join(itens) + "
" def _filtros_html(produtos: list) -> str: """Gera os botões de filtro por categoria (só as categorias presentes).""" cats_presentes = [] for p in produtos: c = _inferir_categoria(p) if c not in cats_presentes: cats_presentes.append(c) if len(cats_presentes) <= 1: return "" # não faz sentido filtro com 1 categoria só botoes = [''] for c in cats_presentes: botoes.append(f'') return '
' + "".join(botoes) + "
" def gerar_pagina(produtos: list) -> str: """Monta o HTML completo da página de links.""" # 1º produto (maior score) vira destaque largo; o resto vai pra grade 2 col. destaque_html = "" grade_cards = [] if produtos: destaque_html = _card_produto(produtos[0], destaque=True) for p in produtos[1:]: grade_cards.append(_card_produto(p, destaque=False)) if produtos: grade_html = ('

" if grade_cards else "") subtitulo_grade = ('

' if grade_cards else "") cards_html = destaque_html + subtitulo_grade + grade_html else: cards_html = '

' redes = _redes_html() filtros = _filtros_html(produtos) ano = time.strftime("%Y") total = len(produtos) return f""" {MARCA} — Achados da Shopee

""" def main(): parser = argparse.ArgumentParser( description="Gera a página de links (bio) estática da TopShop") parser.add_argument("--abrir", action="store_true", help="Abre a página no navegador depois de gerar") parser.add_argument("--sem-shopee", action="store_true", dest="sem_shopee", help="Não tenta gerar links via Shopee (usa só os existentes)") parser.add_argument("--saida", default="", help="Caminho de saída do HTML (default: site/index.html)") args = parser.parse_args() log.info("=" * 60) log.info("🔗 BIO PAGE BUILDER — Página de Links TopShop") log.info("=" * 60) produtos = _carregar_produtos() if not produtos: log.error("❌ Nenhum produto encontrado (rode o validar_fila/curar_fila antes)") return 1 log.info(f"📦 {len(produtos)} produtos carregados") # Gera os links de afiliado (a menos que --sem-shopee) if not args.sem_shopee: log.info("🔗 Gerando links de afiliado (sub-id 'bio' pra rastreio)...") produtos = _gerar_links_afiliado(produtos) else: produtos = [p for p in produtos if p.get("link")] if not produtos: log.error("❌ Nenhum produto com link de afiliado — nada pra publicar") log.info(" (rode com a Shopee configurada, ou garanta links na fila)") return 1 # Ordena: mina_ouro primeiro (maior comissão no topo), depois o resto produtos.sort(key=lambda p: ( 0 if p.get("classe") == "mina_ouro" else 1, -float(p.get("comissao_valor", 0) or 0))) html_final = gerar_pagina(produtos) saida = Path(args.saida) if args.saida else SAIDA_HTML saida.parent.mkdir(parents=True, exist_ok=True) with open(saida, "w", encoding="utf-8") as f: f.write(html_final) log.info(f"✅ Página gerada: {saida}") log.info(f" {len(produtos)} produtos com link | tamanho {len(html_final)//1024}KB") print(f"\n🔗 Página de links pronta: {saida}") print(f" Sobe esse arquivo no GitHub Pages e aponta tua bio pra ele!") if args.abrir: import webbrowser webbrowser.open(f"file://{saida.resolve()}") return 0 if __name__ == "__main__": sys.exit(main())
